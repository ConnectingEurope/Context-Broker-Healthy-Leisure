package com.example.nereoapp.Fragments

import android.os.Bundle
import androidx.fragment.app.Fragment
import android.view.LayoutInflater
import android.view.View
import android.view.ViewGroup
import com.example.nereoapp.R
import kotlinx.android.synthetic.main.fragment_street.*
import org.osmdroid.util.GeoPoint
import org.osmdroid.views.MapController
import org.osmdroid.views.MapView

class StreetFragment : Fragment() {
    private lateinit var myOpenMapView: MapView
    private lateinit var myMapController: MapController
    private var Street = GeoPoint(38.5350154, -0.1171847)


    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        myOpenMapView = openmapview as MapView
        myOpenMapView.setBuiltInZoomControls(true)
        myMapController= myOpenMapView.getController() as MapController
        myMapController.setCenter(Street)
        myMapController.setZoom(6)
        myOpenMapView.setMultiTouchControls(true)

    }
    override fun onCreateView(inflater: LayoutInflater, container: ViewGroup?,
                              savedInstanceState: Bundle?): View? {
        // Inflate the layout for this fragment
        return inflater.inflate(R.layout.fragment_street, container, false)
    }
}
